// Demo 4: Pagination Style
let currentPage = 1;
const totalPages = 5;

document.querySelectorAll(".page-btn").forEach((btn) => {
  btn.addEventListener("click", function () {
    const targetPage = parseInt(this.dataset.page);
    showPage(targetPage);
  });
});

document.getElementById("nextPage").addEventListener("click", function () {
  if (currentPage < totalPages) {
    showPage(currentPage + 1);
  }
});

function showPage(pageNum) {
  // Hide all pages
  document.querySelectorAll(".page-content").forEach((page) => {
    page.style.display = "none";
  });

  // Show target page
  const targetPage = document.querySelector(`.page-${pageNum}`);
  if (targetPage) {
    targetPage.style.display = "block";
    targetPage.classList.add("fade-up");
  }

  // Update active button
  document.querySelectorAll(".page-btn").forEach((btn) => {
    btn.classList.remove("active");
    if (parseInt(btn.dataset.page) === pageNum) {
      btn.classList.add("active");
    }
  });

  // Update next button
  const nextBtn = document.getElementById("nextPage");
  if (pageNum >= totalPages) {
    nextBtn.disabled = true;
    nextBtn.innerHTML = 'Last Page <i class="fas fa-check ms-1"></i>';
  } else {
    nextBtn.disabled = false;
    nextBtn.innerHTML = 'Next <i class="fas fa-arrow-right ms-1"></i>';
  }

  currentPage = pageNum;
}
